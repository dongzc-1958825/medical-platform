// src/shared/types/base.ts - 最基础的公共类型
export interface Identifiable {
  id: string;
}

export interface Timestamped {
  createdAt: string; // ISO 8601格式
  updatedAt?: string;
}

// 数据可见性级别
export enum DataVisibility {
  PUBLIC = 'public',           // 完全公开
  PROTECTED = 'protected',     // 登录用户可见
  PRIVATE = 'private',         // 仅自己可见
  ENCRYPTED = 'encrypted'      // 加密存储，严格隐私
}
